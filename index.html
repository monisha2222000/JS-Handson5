Q.2 Count the Heads

        var Count_Occ = (s) => {
            let obj = {};
            let str = "";
            let r = s.split("").sort();
            for (let i = 0; i < r.length; i++) {
                if (obj[r[i]]) {
                    obj[r[i]]++;
                } else {
                    obj[r[i]] = 1;
                }
            }
            for (let key in obj) {
                if (obj[key] > 1) {
                    str += key + obj[key];
                }
            }
            return str; 
        };

Q.8 Plaindrome Check

        var Palin_Check = (str) => {
            let rev = "";
            for (let i=str.length-1; i>=0; i--){
                rev += str[i];
            }
            if (rev == str){
                return "True";
            }
            else
            return "False";
        };

        var Reverse_String = (str) => {
